export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.invert-0': { '--tw-invert': 'invert(0)' },
        '.invert': { '--tw-invert': 'invert(1)' },
      })
    }
  }
}

## Usage

Use the `invert` and `invert-0` utilities alongside the `filter` utility to whether an element should be rendered with inverted colors or normally.

```html lightBlue
<template preview>
  <div class="flex justify-around">
    <div class="flex-1">
      <p class="text-center text-sm text-light-blue-600 font-medium mb-1">.invert-0</p>
      <div class="mx-auto h-24 w-24 rounded-xl shadow-md bg-light-blue-400 filter invert-0">
      </div>
    </div>
    <div class="flex-1">
      <p class="text-center text-sm text-light-blue-600 font-medium mb-1">.invert</p>
      <div class="mx-auto h-24 w-24 rounded-xl shadow-md bg-light-blue-400 filter invert">
      </div>
    </div>
    <div class="flex-1"></div>
  </div>
</template>

<div class="filter **invert-0** ..."></div>
<div class="filter **invert** ..."></div>
```

## Responsive

To control an element's invert filter at a specific breakpoint, add a `{screen}:` prefix to any existing invert utility. For example, use `md:invert-0` to apply the `invert-0` utility at only medium screen sizes and above.

```html
<div class="filter invert **md:invert-0** ...">
  <!-- ... -->
</div>
```

For more information about Tailwind's responsive design features, check out the [Responsive Design](/docs/responsive-design) documentation.
